busca en tu base de datos qué usuarios tienen actualmente el rol de Admin (nuestro Spatie RBAC usa Admin, no super_admin, según el Codex)

SELECT u.id, u.full_name, u.email, r.name as role_name
FROM users u
JOIN model_has_roles mhr ON u.id = mhr.model_id AND mhr.model_type = 'App\\User'
JOIN roles r ON mhr.role_id = r.id
WHERE r.name = 'Admin';
